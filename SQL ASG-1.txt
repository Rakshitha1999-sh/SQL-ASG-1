1.	SELECT EMPNO, ENAME, JOB
	FROM EMP
	WHERE JOB LIKE 'CLERK';

2.	SELECT EMPNO, ENAME, JOB
	FROM EMP
	WHERE JOB IN ('CLERK', 'MANAGER');

3.	SELECT CURRENT_TIMESTAMP;

4.	SELECT ENAME, SAL
	FROM EMP
	WHERE SAL BETWEEN 500 AND 1000;

5.	SELECT *
	FROM EMP
	WHERE HIREDATE LIKE '%DEC%';

6.	SELECT ENAME, SAL
	FROM EMP
	WHERE SAL NOT BETWEEN 1000 AND 2500;

7.	SELECT ENAME, SAL
	FROM EMP
	WHERE SAL IN (800, 950, 3000, 5000);

8.	SELECT ENAME
	FROM EMP
	WHERE ENAME LIKE '_L%';

9.	SELECT ENAME
	FROM EMP
	WHERE ENAME LIKE '%D';

10.	SELECT ENAME
	FROM EMP
	WHERE LENGTH(ENAME) = 5;

11.	SELECT ENAME
	FROM EMP
	WHERE ENAME LIKE 'T%R';

12.	SELECT ENAME
	FROM EMP
	WHERE ENAME LIKE '%A%';

13.	SELECT ENAME, HIREDATE
	FROM EMP
	WHERE TO_CHAR(HIREDATE,'DD') < 19;

14.	SELECT EMPNO, ENAME
	FROM EMP
	WHERE ENAME = 'JAMES';

15.	SELECT EMPNO, ENAME, SAL
	FROM EMP
	WHERE ENAME = 'KING';

16.	SELECT ENAME, JOB
	FROM EMP
	WHERE JOB = 'PRESIDENT';

17.	SELECT *
	FROM EMP
	WHERE JOB = 'MANAGER';

18.	SELECT *
	FROM EMP
	WHERE COMM IS NOT NULL;

19.	SELECT ENAME, JOB, HIREDATE
	FROM EMP
	WHERE HIREDATE < ('01-APR-81');

20.	SELECT ENAME, SAL
	FROM EMP
	WHERE ENAME IN ('KING','BLAKE','FORD','SMITH');

22.	SELECT ENAME, SAL
	FROM EMP
	WHERE SAL < 3500;

23.	SELECT *
	FROM EMP
	WHERE JOB = 'CLERK';

24.	SELECT ENAME, SAL AS 'MONTHLY SALARY', SAL*12 AS 'ANNUAL SALARY'
	FROM EMP;

25.	SELECT ENAME, SAL, SAL+(SAL*20/100) AS 'HIKED_SALARY', (SAL+(SAL*20/100))*12 AS 'ANNUAL SALARY'
	FROM EMP
	WHERE ENAME='FORD';

26.	SELECT E.DEPTNO, D.DNAME, SUM(SAL) AS 'TOTAL SALARY'
	FROM EMP E, DEPT D
    	WHERE E.DEPTNO = D.DEPTNO
	GROUP BY E.DEPTNO;

27.	SELECT ENAME, MIN(SAL) AS 'LOWEST SALARY'
	FROM EMP
	WHERE DEPTNO = 10;

28.	SELECT JOB, ROUND(AVG(SAL),2) AS 'AVERAGE SALARY'
	FROM EMP
	WHERE JOB = 'MANAGER';

29.	SELECT JOB, SUM(SAL) AS 'TOTAL SALARY'
	FROM EMP
	WHERE JOB = 'SALESMAN';

30.	SELECT DEPTNO, AVG(SAL) AS 'LOW_AVG_SALARY'
	FROM EMP
	GROUP BY DEPTNO
	HAVING AVG(SAL)< (SELECT AVG(SAL) FROM EMP);

31.	SELECT JOB, COUNT(*) AS 'NO_OF_PERSONS'
	FROM EMP
	GROUP BY JOB;

32.	SELECT JOB, MAX(SAL) AS 'HIGHEST SALARY'
	FROM EMP
	WHERE JOB = 'ANALYST';

33.	SELECT ENAME, MAX(COMM) AS 'HIGHEST_COMMISSION'
	FROM EMP;

34.	SELECT COUNT(DEPTNO) AS 'NO_OF_DEPARTMENTS'
	FROM DEPT;

35.	SELECT ENAME
	FROM EMP
	WHERE ENAME LIKE '__L%';
